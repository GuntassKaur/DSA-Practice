import java.util.*;

public class Solution {

    static int[] gradingStudents(int[] grades) {
        int[] result = new int[grades.length];

        for (int i = 0; i < grades.length; i++) {
            int g = grades[i];

            // No rounding if grade is below 38
            if (g < 38) {
                result[i] = g;
                continue;
            }

            int nextMultiple = ((g / 5) + 1) * 5;
            int diff = nextMultiple - g;

            if (diff < 3) {
                result[i] = nextMultiple;
            } else {
                result[i] = g;
            }
        }

        return result;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        int[] grades = new int[n];
        for (int i = 0; i < n; i++) {
            grades[i] = sc.nextInt();
        }

        int[] finalGrades = gradingStudents(grades);

        for (int g : finalGrades) {
            System.out.println(g);
        }
        
        sc.close();
    }
}
